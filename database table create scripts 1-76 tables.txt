(
    user_id bigint NOT NULL GENERATED ALWAYS AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1 ),

    name_bn character varying(100) NOT NULL,
    username character varying(100) NOT NULL,
    mobile character varying(100) NOT NULL,
    email character varying(255) DEFAULT NULL,
    password character varying(255),
    user_type_id integer,
    designation_id integer,
    profile_completed boolean NOT NULL DEFAULT false,
    is_active boolean DEFAULT true,
    create_user_id bigint,
    update_user_id bigint,
    created_date timestamp without time zone NOT NULL,
    updated_date timestamp without time zone,
    deleted_date timestamp without time zone,
    version_no smallint NOT NULL DEFAULT '1',
    CONSTRAINT s_users_pkey PRIMARY KEY (user_id)
);
